        }
       
        function setup() {
          // shaders require WEBGL mode to work
          createCanvas(710, 400, WEBGL);
          noStroke();
       
          cam = createCapture(VIDEO);
          cam.size(710, 400);

        }
       
        function draw() {
          // shader() sets the active shader with our shader
          shader(theShader);
