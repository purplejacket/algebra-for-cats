14. Design an algorithm then write a function collatzProof(num) that will prove the Collatz Conjecture for the integers from 1 through num. Here's a helper question: What might the output of collatzProof(4) look like?


  function collatz(n){

    var results = [];

    for (i = 1; i<=n; i++){

        var within = [];

        var result = i;

        if(result == 1){

            within.push([1]);

        }else {

            while (result != 1){

                if (result%2 == 0){

                    result = result/2;

                } else{

                    result = result*3+1;

                }

                within.push(result);

            }

        }

        results.push(within);

    }

    return results;

}
